def send()
    @session.print "HTTP/1.1 #{@status_code}\r\n"
    unless @headers.empty?
      @headers.each do |name, value|
        @session.print "#{name}: #{value}\r\n"
      end
    end
    @session.print "\r\n"
    @session.print @body unless @body.empty?
    @session.close
end
