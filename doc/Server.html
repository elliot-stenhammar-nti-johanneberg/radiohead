def start()
    server = TCPServer.new(@port)

    while session = server.accept
        data = ""

        while line = session.gets and line !~ /^\s*$/
            data += line
        end

        request = Request.new(data)
        response = ResponseBuilder.new(request, session, @router).build()
        response.send()
    end
end
